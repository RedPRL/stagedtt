module D := Data
type global = [
| `Unstaged of Prelude.Ident.path * Data.value Stdlib.Lazy.t * Data.inner Stdlib.Lazy.t
| `Staged of Prelude.Ident.path * Data.value Stdlib.Lazy.t * Data.inner Stdlib.Lazy.t * ( int -> Data.outer )
]
and t = Data.syntax =
| Local of int
| Global of global
| Lam of Prelude.Ident.t * t
| Ap of t * t
| Quote of t
| Splice of t
| CodePi of t * t
| CodeUniv of int
