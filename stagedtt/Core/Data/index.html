Syntax

type global = [
| `Unstaged of Prelude.Ident.path * value Stdlib.Lazy.t * inner Stdlib.Lazy.t
| `Staged of Prelude.Ident.path * value Stdlib.Lazy.t * inner Stdlib.Lazy.t * ( int -> outer )
]
and syntax =
| Local of int
| Global of global
| Lam of Prelude.Ident.t * syntax
| Ap of syntax * syntax
| Quote of syntax
| Splice of syntax
| CodePi of syntax * syntax
| CodeUniv of int
and syntax_tp =
| TpVar of int(*

DeBruijin-Indexed type variables. These are used during grafting.

*)
| Pi of syntax_tp * Prelude.Ident.t * syntax_tp
| Expr of syntax_tp
| El of syntax
| Univ of int

Values

and value =
| Lam of Prelude.Ident.t * tm_vclo
| Quote of value
| Neu of neu
| Code of code
and code =
| CodePi of value * value
| CodeUniv of int
and value_tp =
| Pi of value_tp * Prelude.Ident.t * tp_vclo
| Expr of value_tp
| El of code
| ElNeu of neu
| Univ of int
and neu = {
hd : hd;
spine : frm list;
}
and hd =
| Local of int
| Global of [ `Unstaged of Prelude.Ident.path * value Stdlib.Lazy.t * inner Stdlib.Lazy.t ]
and frm =
| Ap of value
| Splice
and 'a vclo =
| Clo of 'a * value_env
and tp_vclo = syntax_tp vclo
and tm_vclo = syntax vclo
and value_env = {
locals : value Stdlib.Lazy.t Prelude.bwd;
size : int;
tp_locals : value_tp Prelude.bwd;
tp_size : int;
}

Staging

and outer =
| Lam of Prelude.Ident.t * tm_sclo
| Quote of inner
| Irrelevant
and inner =
| Local of int
| Global of global
| Lam of Prelude.Ident.t * inner
| Ap of inner * inner
| Quote of inner
| Splice of inner
| CodePi of inner * inner
| CodeUniv of int
and staged =
| Inner of inner
| Outer of outer
and stage_env = {
locals : staged Prelude.bwd;
size : int;
}
and 'a sclo =
| Clo of 'a * stage_env
and tm_sclo = syntax sclo
